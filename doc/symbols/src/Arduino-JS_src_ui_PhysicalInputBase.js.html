  4 
  5 ARDJS.namespace('ARDJS.ui.PhysicalInputBase');
  6 
  7 ARDJS.ui.PhysicalInputBase = (function() {
  8 
  9 	var PhysicalInputBase;
 10 
 11 	// dependencies
 12 	var EventDispatcher = ARDJS.EventDispatcher;
 13 
 14 	/**
 15 	 * A base class for physical inputs. Treat this class as an abstract base
 16 	 * class - do not instantiate it directly.
 17 	 *
 18 	 * @exports PhysicalInputBase as ARDJS.ui.PhysicalInputBase
 19 	 * @constructor
 20 	 */
 21 	PhysicalInputBase = function() {
 22 
 23 		this.name = "PhysicalInputBase"; // for testing
 24 
 25 		this._evtDispatcher = new EventDispatcher(this);
 26 	}
 27 
 28 	PhysicalInputBase.prototype = {
 29 		/* implement EventDispatcher */
 30 		
 31 		/**
 32 		 * @param {String} type The event type
 33 		 * @param {Function} listener The function to be called when the event is fired
 34 		 */
 35 		addEventListener: function(type, listener) {
 36 			this._evtDispatcher.addEventListener(type, listener);
 37 		},
 38 		
 39 		/**
 40 		 * @param {String} type The event type
 41 		 * @param {Function} listener The function to be called when the event is fired
 42 		 */
 43 		removeEventListener: function(type, listener) {
 44 			this._evtDispatcher.removeEventListener(type, listener);
 45 		},
 46 		
 47 		/**
 48 		 * @param {String} type The event type
 49 		 * return {boolean} True is listener exists for this type, false if not.
 50 		 */
 51 		hasEventListener: function(type) {
 52 			return this._evtDispatcher.hasEventListener(type);
 53 		},
 54 		
 55 		/**
 56 		 * @param {Event} type The Event object
 57 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
 58 		 * return {boolean} True if dispatch is successful, false if not.
 59 		 */		
 60 		dispatchEvent: function(event, optionalParams) {
 61 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
 62 		}			
 63 	};
 64 
 65 	return PhysicalInputBase;
 66 
 67 }());
