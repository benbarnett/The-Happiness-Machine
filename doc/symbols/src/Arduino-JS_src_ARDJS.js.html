  6 /** @namespace Namespace and utility functions for Arduino-JS */
  7 var ARDJS = ARDJS || {};

  9 /**
 10  * use this function to safely create a new namespace
 11  * if a namespace already exists, it won't be recreated
 12  *
 13  * @function
 14  */
 15 ARDJS.namespace = function (ns_string) {
 16 	var parts = ns_string.split('.'),
 17 		parent = ARDJS,
 18 		i;
 19 		
 20 	// strip redundant leading global
 21 	if (parts[0] === "ARDJS") {
 22 		parts = parts.slice(1);
 23 	}
 24 	
 25 	for (i=0; i<parts.length; i +=1) {
 26 		// create a property if it doesn't exist
 27 		if (typeof parent[parts[i]] === "undefined") {
 28 			parent[parts[i]] = {};
 29 		}
 30 		parent = parent[parts[i]];
 31 	}
 32 	return parent;
 33 };

 35 /**
 36  * copied from Flanagan, David. JavaScript: The Definitive Guide
 37  *
 38  * Use this method rather than Object.create() directly if
 39  * browser compatibility is unknow
 40  * @function
 41  */
 42 ARDJS.inherit = function(p) {
 43 	if (p == null) throw TypeError(); // p must be a non-null object
 44 	if (Object.create) { // If Object.create() is defined...
 45 		return Object.create(p); // then just use it
 46 	}
 47 	var t = typeof p; // otherwise do some more type checking
 48 	if (t !== "object" && t !== "function") throw TypeError();
 49 	function f() {}; // define a dummy constructor function
 50 	f.prototype = p; // Set its prototype property to p
 51 	return new f(); // use f() to create an 'heir' of p.
 52 }
