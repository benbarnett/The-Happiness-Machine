180 		/**
181 		 * @private
182 		 */
183 		function processData(inputData) {
184 			inputData *= 1;	// force inputData to integer (is there a better way to do this?)
185 			var command;
186 			
187 			// we have command data
188 			if (_waitForData > 0 && inputData < 128) {
189 				_waitForData--;
190 				// collect the data
191 				_storedInputData[_waitForData]=inputData;
192 				// we have all data executeMultiByteCommand
193 				if (_waitForData == 0) {
194 					switch (_executeMultiByteCommand) {
195 						case DIGITAL_MESSAGE:
196 							processDigitalPortBytes(_multiByteChannel, _storedInputData[1], _storedInputData[0]); //(LSB, MSB)
197 							break;
198 						case REPORT_VERSION:
199 							_firmwareVersion=_storedInputData[1] + _storedInputData[0] / 10;
200 							self.dispatchEvent(new ArduinoEvent(ArduinoEvent.FIRMWARE_VERSION), {version: _firmwareVersion});
201 							break;
202 						case ANALOG_MESSAGE:
203 							var analogPin = self.getAnalogPin(_multiByteChannel);
204 							// NOTE: is there a better way to handle this? This issue is on browser refresh
205 							// the Arduino board is still sending analog data if analog reporting was set
206 							// before the refresh. Analog reporting won't be disabled by systemReset systemReset()
207 							// is called. There is not way to call that method fast enough so the following code is
208 							// needed. An alternative would be to set a flag that prevents critical operations
209 							// before systemReset has completed
210 							if (analogPin == undefined) {
211 								//console.log("analog pin undefined");
212 								break;
213 							}
214 							// map analog input values from 0-1023 to 0.0 to 1.0
215 							analogPin.value = self.getValueFromTwo7bitBytes(_storedInputData[1], _storedInputData[0])/1023;
216 							if (analogPin.value != analogPin.lastValue) {
217 								// use analog pin number rather than actual pin number
218 								self.dispatchEvent(new ArduinoEvent(ArduinoEvent.ANALOG_DATA), {pin: _multiByteChannel, value: analogPin.value});
219 							}
220 							break;
221 					}
222 				}
223 			}
224 			// we have SysEx command data
225 			else if (_waitForData < 0) {
226 				// we have all sysex data
227 				if (inputData == END_SYSEX) {
228 					_waitForData = 0;
229 					switch (_sysExData[0]) {
230 						case REPORT_FIRMWARE:
231 							processQueryFirmwareResult(_sysExData);
232 							break;
233 						case STRING_DATA:
234 							processSysExString(_sysExData);
235 							break;
236 						case CAPABILITY_RESPONSE:
237 							processCapabilitiesResponse(_sysExData);
238 							break;
239 						case PIN_STATE_RESPONSE:
240 							processPinStateResponse(_sysExData);
241 							break;
242 						case ANALOG_MAPPING_RESPONSE:
243 							processAnalogMappingResponse(_sysExData);
244 							break;
245 						default:
246 							// custom sysEx message
247 							self.dispatchEvent(new ArduinoEvent(ArduinoEvent.SYSEX_MESSAGE), {message: _sysExData});
248 							break;
249 					}
250 					_sysExData = [];
251 				}
252 				else {
253 					_sysExData.push(inputData);
254 				}
255 			}
256 			// we have a command
257 			else {
258 				// extract the command and channel info from a byte if it is less than 0xF0
259 				if (inputData < 240) {
260 					command = inputData & 240;
261 					_multiByteChannel = inputData & 15;
262 				}
263 				else {
264 					// commands in the 0xF* range don't use channel data
265 					command = inputData;
266 				}
267 				switch (command) {
268 					case REPORT_VERSION:
269 					case DIGITAL_MESSAGE:
270 					case ANALOG_MESSAGE:
271 						_waitForData = 2;	// 2 bytes needed
272 						_executeMultiByteCommand = command;
273 						break;
274 					case START_SYSEX:
275 						_waitForData = -1; // n bytes needed
276 						_executeMultiByteCommand = command;
277 						break;
278 					default:
279 						break;
280 				}
281 			}
282 		}

287 		function processDigitalPortBytes(port, bits0_6, bits7_13) {
288 			var offset = port * 8;
289 			var lastPin = offset + 8;
290 			var portVal = bits0_6 | (bits7_13 << 7);
291 			var pinVal;
292 			var pin = {};
293 			
294 			if (lastPin >= _totalPins) lastPin = _totalPins;
295 			
296 			var j=0;
297 			for (var i=offset; i<lastPin; i++) {
298 				pin = self.getDigitalPin(i);
299 				// ignore data send on Firmata startup
300 				// same for analog?
301 				if (pin == undefined) return;
302 				
303 				if (pin.type == Pin.DIN) {
304 					pinVal = (portVal >> j) & 0x01;	// test this
305 		    		if (pinVal != pin.value) {
306 		    			pin.value = pinVal;
307 		    			self.dispatchEvent(new ArduinoEvent(ArduinoEvent.DIGITAL_DATA), {pin: pin.number, value: pin.value});
308 		    		}
309 		    	}
310 		    	j++;
311 		    }
312 		}

345 		/** 
346 		 * Auto configure using capabilities response
347 		 * This should create a configuration for any board in the Firmata boards.h file
348 		 *
349 		 * @private
350 		 */
351 		function processCapabilitiesResponse(msg) {
352 			var pinCapabilities = {};
353 			var byteCounter = 1; // skip 1st byte because it's the command
354 			var pinCounter = 0;
355 			var analogPinCounter = 0;
356 			var len = msg.length;
357 			var type;
358 					
359 			// create default configuration
360 			while (byteCounter <= len) {
361 				// 127 denotes end of pin's modes
362 				if (msg[byteCounter] == 127) {
363 					
364 					// is digital pin mapping even necessary anymore?
365 					_digitalPinMapping[pinCounter] = pinCounter;
366 					type = undefined;
367 					
368 					// assign default types
369 					if (pinCapabilities[Pin.DOUT]) {
370 						// map digital pins
371 						type = Pin.DOUT;
372 					}
373 					
374 					if (pinCapabilities[Pin.AIN]) {
375 						type = Pin.AIN;
376 						// map analog input pins
377 						_analogPinMapping[analogPinCounter++] = pinCounter;
378 					} 
379 					
380 					var pin = new Pin(pinCounter, type);
381 					pin.capabilities = pinCapabilities;
382 					managePinListener(pin);
383 					_ioPins[pinCounter] = pin;
384 					
385 					// store the 2 i2c pin numbers if they exist
386 					if (pin.capabilities[Pin.I2C]) {
387 						_i2cPins.push(pin.number);
388 					}
389 					
390 					pinCapabilities = {};
391 					pinCounter++;
392 					byteCounter++;
393 				} else {
394 					// create capabilities object (mode: resolution) for each  mode
395 					// supported by each pin
396 					pinCapabilities[msg[byteCounter]] = msg[byteCounter + 1];
397 					byteCounter += 2;
398 				}
399 			}
400 			
401 			_numPorts = Math.ceil(pinCounter / 8);
402 			console.log("debug: num ports = " + _numPorts);
403 			
404 			// initialize port values
405 			for (var j=0; j<_numPorts; j++) {
406 				_digitalPort[j] = 0;
407 			}
408 			
409 			_totalPins = pinCounter;
410 			console.log("debug: num pins = " + _totalPins);
411 			
412 			self.reportCapabilities();
413 
414 			console.log("debug: system reset");
415 			systemReset();
416 
417 			// delay to allow systemReset function to execute in StandardFirmata
418 			setTimeout(startup, 500);
419 			console.log("debug: configured");
420 		}

495 		/**
496 		 * Called when ever a pin value is set via pin.setValue(someValue).
497 		 * Sends digital or analog output pin and output values to the Arduino.
498 		 *
499 		 * @private
500 		 * @param {Event} pin A reference to the event object (Pin in this case).
501 		 */
502 		 function sendOut(event) {
503 		 	var type = event.target.type;
504 		 	var pinNum = event.target.number;
505 		 	var value = event.target.value;
506 		 	
507 		 	switch(type) {
508 		 		case Pin.DOUT:
509 		 			sendDigitalData(pinNum, value);
510 		 		break;
511 		 		case Pin.AOUT:
512 		 			sendAnalogData(pinNum, value);
513 		 		break;
514 		 		case Pin.SERVO:
515 		 			sendServoData(pinNum, value);
516 		 		break;
517 		 	}
518 
519 		 }

521 		/**
522 		 * Ensure that event listeners are properly managed for pin objects as the pin
523 		 * type is changed during the execution of the program.
524 		 *
525 		 * @private
526 		 */	 
527 		function managePinListener(pin) {
528 			if (pin.type == Pin.DOUT || pin.type == Pin.AOUT || pin.type == Pin.SERVO) {
529 				if (!pin.hasEventListener(Event.CHANGE)) {
530 					pin.addEventListener(Event.CHANGE, sendOut);
531 				}
532 			} else {
533 				if (pin.hasEventListener(Event.CHANGE)) {
534 					try {
535 						pin.removeEventListener(Event.CHANGE, sendOut);
536 					} catch (e) {
537 						// pin had reference to other handler, ignore
538 						console.log("debug: caught pin removeEventListener exception");
539 					}
540 				}
541 			}
542 		}

545 		 * @private
546 		 */
547 		function sendAnalogData(pin, value) {
548 			if (pin > 15 || value > Math.pow(2, 14)) {
549 				sendExtendedAnalogData(pin, value);
550 			} else {
551 				self.send([ANALOG_MESSAGE | (pin & 0x0F), value & 0x007F,(value >> 7) & 0x007F]);
552 			}
553 		}

555 		/**
556 		 * @private
557 		 */	
558 		function sendExtendedAnalogData(pin, value) {
559 			var tempArray = [];
560 			
561 			// if > 16 bits
562 			if (value > Math.pow(2, 16)) {
563 				var err = "Extended Analog values > 16 bits are not currently supported by StandardFirmata";
564 				console.log(err);
565 				throw err;
566 				return;
567 			}
568 			
569 			tempArray[0] = START_SYSEX;
570 			tempArray[1] = EXTENDED_ANALOG;
571 			tempArray[2] = pin;
572 			tempArray[3] = value & 0x007F;
573 		 	tempArray[4] = (value >> 7) & 0x007F;	// up to 14 bits
574 					
575 		 	// if > 14 bits
576 		 	if (value >= Math.pow(2, 14)) {
577 		 		tempArray[5] = (value >> 14) & 0x007F;
578 		 	}
579 		 	
580 			tempArray.push(END_SYSEX);
581 			self.send(tempArray);
582 		}

584 		/**
585 		 * @private
586 		 */
587 		function sendDigitalData(pin, value) {
588 			var portNum = Math.floor(pin / 8);
589 
590 			if (value == Pin.HIGH) {
591 				// set the bit
592 				_digitalPort[portNum] |= (value << (pin % 8));
593 			}
594 			else if (value == Pin.LOW) {
595 				// clear the bit
596 				_digitalPort[portNum] &= ~(1 << (pin % 8));
597 			}
598 			else {
599 				console.log("invalid value passed to sendDigital, value must be 0 or 1");
600 				return; // invalid value
601 			}
602 			
603 			self.sendDigitalPort(portNum, _digitalPort[portNum]);	
604 		}

627 		/**
628 		 * A utility class to assemble a single value from the 2 bytes returned from the
629 		 * Arduino (since data is passed in 7 bit Bytes rather than 8 bit it must be 
630 		 * reassembled. This is to be used as a protected method and should not be needed
631 		 * in any application level code.
632 		 *
633 		 * @private
634 		 * @param {Number} lsb The least-significant byte of the 2 values to be concatentated
635 		 * @param {Number} msb The most-significant byte of the 2 values to be concatenated
636 		 * @return {Number} The result of merging the 2 bytes
637 		 */
638 		this.getValueFromTwo7bitBytes = function(lsb, msb) {
639 			return (msb <<7) | lsb;
640 		}

904 			var tempArray = [];
905 			tempArray[0] = START_SYSEX;

909 			tempArray[4] = minPulse >> 7;
