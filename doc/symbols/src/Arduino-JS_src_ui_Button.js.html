  5 ARDJS.namespace('ARDJS.ui.ButtonEvent');
  6 
  7 ARDJS.ui.ButtonEvent = (function() {
  8 
  9 	var ButtonEvent;
 10 
 11 	// dependencies
 12 	var Event = ARDJS.Event;
 13 
 14 	/**
 15 	 * @exports ButtonEvent as ARDJS.ui.ButtonEvent
 16 	 * @constructor
 17 	 * @augments ARDJS.Event
 18 	 */
 19 	ButtonEvent = function(type) {
 20 
 21 		this.name = "ButtonEvent"; // for testing
 22 
 23 		Event.call(this, type);
 24 	}
 25 
 26 	/** @constant */
 27 	ButtonEvent.PRESS = "pressed";
 28 	/** @constant */
 29 	ButtonEvent.RELEASE = "released";
 30 	/** @constant */
 31 	ButtonEvent.LONG_PRESS = "longPress";
 32 	/** @constant */
 33 	ButtonEvent.SUSTAINED_PRESS = "sustainedPress";
 34 
 35 	ButtonEvent.prototype = ARDJS.inherit(Event.prototype);
 36 	ButtonEvent.prototype.constructor = ButtonEvent;
 37 
 38 	return ButtonEvent;
 39 
 40 }());

 43 ARDJS.namespace('ARDJS.ui.Button');
 44 
 45 ARDJS.ui.Button = (function() {
 46 
 47 	var Button;
 48 
 49 	// dependencies
 50 	var PhysicalInputBase = ARDJS.ui.PhysicalInputBase,
 51 		Event = ARDJS.Event,
 52 		ButtonEvent = ARDJS.ui.ButtonEvent;
 53 
 54 	/**
 55 	 * An object to represent a physical button. The advantage of using the Button class over
 56 	 * listening for pin change events on a Pin object, is that the Button class handles debouncing 
 57 	 * and provides helpful button events: Pressed, Released, Long Press and Sustained Press
 58 	 *
 59 	 * @exports Button as ARDJS.ui.Button
 60 	 * @constructor
 61 	 * @augments ARDJS.ui.PhysicalInputBase
 62 	 * @param {Pin} pin A reference to the pin the button is connected to.
 63 	 * @param {number} buttonMode The mode of the button (either PULL_DOWN or PULL_UP). 
 64 	 * Default is PULL_DOWN.
 65 	 * @param {number} longPressDelay The delay time in milliseconds before a long press event is fired.
 66 	 *
 67 	 */
 68 	Button = function(pin, buttonMode, longPressDelay) {
 69 		"use strict";
 70 		
 71 		PhysicalInputBase.call(this);
 72 
 73 		this.name = "Button"; // for testing
 74 		this._pin = pin;
 75 		
 76 		this.buttonMode = buttonMode || Button.PULL_DOWN;
 77 		this.longPressDelay = longPressDelay || 1000;
 78 
 79 		this._debounceInterval = 20,
 80 		this._repeatCount = 0,
 81 		this._timer = null,
 82 		this._timeout = null;	
 83 		
 84 		pin.addEventListener(Event.CHANGE, this.onPinChange.bind(this));
 85 		
 86 	}
 87 
 88 
 89 	Button.prototype = ARDJS.inherit(PhysicalInputBase.prototype);
 90 	Button.prototype.constructor = Button;
 91 
 92 
 93 	/**
 94 	 * @private
 95 	 */
 96 	Button.prototype.onPinChange = function(evt) {
 97 		
 98 		var btnVal = evt.target.value;
 99 		var stateHandler;
100 
101 		if (this.buttonMode === Button.PULL_DOWN) {
102 			if (btnVal === 1) stateHandler = this.pressed;
103 			else stateHandler = this.released;
104 		} else if (this.buttonMode === Button.PULL_UP) {
105 			if (btnVal === 1) stateHandler = this.released;
106 			else stateHandler = this.pressed;
107 		}
108 		
109 		if (this._timeout === null) {
110 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
111 		} else {
112 			clearTimeout(this._timeout);
113 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
114 		}
115 	};
116 	
117 	/**
118 	 * @private
119 	 */
120 	Button.prototype.pressed = function() {
121 		this._timeout = null;
122 
123 		this.dispatchEvent(new ButtonEvent(ButtonEvent.PRESS));
124 		
125 		this_timer = setInterval(this.sustainedPress.bind(this), this.longPressDelay);
126 	};
127 	
128 	/**
129 	 * @private
130 	 */	
131 	Button.prototype.released = function() {
132 		this._timeout = null;
133 		this.dispatchEvent(new ButtonEvent(ButtonEvent.RELEASE));
134 		
135 		if (this._timer != null) {
136 			clearInterval(this._timer);
137 			this._timer = null;
138 		}
139 		
140 		this._repeatCount = 0;
141 	};
142 	
143 	/**
144 	 * @private
145 	 */
146 	Button.prototype.sustainedPress = function() {
147 		if (this._repeatCount > 0) {
148 			this.dispatchEvent(new ButtonEvent(ButtonEvent.SUSTAINED_PRESS));
149 		} else {
150 			this.dispatchEvent(new ButtonEvent(ButtonEvent.LONG_PRESS));
151 		}
152 		
153 		this._repeatCount++;
154 	};
155 	
156 	/**
157 	 * The debounce time interval in milliseconds.
158 	 * @name Button#debounceInterval
159 	 * @property
160 	 * @type Number
161 	 */ 
162 	Button.prototype.__defineGetter__("debounceInterval", function() { return this._debounceInterval; });
163 	Button.prototype.__defineSetter__("debounceInterval", function(interval) { this._debounceInterval = interval; });
164 	
165 	/**
166 	 * [read-only] The pin number of the pin the button is attached to.
167 	 * @name Button#pinNumber
168 	 * @property
169 	 * @type Number
170 	 */
171 	Button.prototype.__defineGetter__("pinNumber", function() { return this._pin.number; });	
172 
173 	/** @constant */
174 	Button.PULL_DOWN = 0;
175 	/** @constant */
176 	Button.PULL_UP 	= 1;
177 
178 	return Button;
179 
180 }());
