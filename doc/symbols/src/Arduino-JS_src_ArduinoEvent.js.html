  5 ARDJS.namespace('ARDJS.ArduinoEvent');
  6 
  7 ARDJS.ArduinoEvent = (function() {
  8 
  9 	var ArduinoEvent;
 10 
 11 	// dependencies
 12 	var Event = ARDJS.Event;
 13 
 14 	/**
 15 	 * @exports ArduinoEvent as ARDJS.ArduinoEvent
 16 	 * @constructor
 17 	 * @augments ARDJS.Event
 18 	 * @param {String} type The event type
 19 	 * @param {Object} data An object containing additional parameters
 20 	 */
 21 	ArduinoEvent = function(type) {
 22 
 23 		this.name = "ArduinoEvent"; // for testing
 24 		
 25 		// call the super class
 26 		// 2nd parameter is passed to EventDispatcher constructor
 27 		Event.call(this, type);
 28 	}
 29 
 30 	// events
 31 	/** @constant */
 32 	ArduinoEvent.ANALOG_DATA = "analodData";
 33 	/** @constant */
 34 	ArduinoEvent.DIGITAL_DATA = "digitalData";
 35 	/** @constant */
 36 	ArduinoEvent.FIRMWARE_VERSION = "firmwareVersion";
 37 	/** @constant */
 38 	ArduinoEvent.FIRMWARE_NAME = "firmwareName";
 39 	/** @constant */
 40 	ArduinoEvent.STRING_MESSAGE = "stringMessage";
 41 	/** @constant */
 42 	ArduinoEvent.SYSEX_MESSAGE = "sysexMessage";
 43 	/** @constant */
 44 	ArduinoEvent.CAPABILITY_RESPONSE = "capabilityResponse";
 45 	/** @constant */
 46 	ArduinoEvent.PIN_STATE_RESPONSE = "pinStateResponse";
 47 	/** @constant */
 48 	ArduinoEvent.ANALOG_MAPPING_RESPONSE = "analogMappingResponse";
 49 	/** @constant */
 50 	ArduinoEvent.READY = "arduinoReady";
 51 
 52 
 53 	ArduinoEvent.prototype = ARDJS.inherit(Event.prototype);
 54 	ArduinoEvent.prototype.constructor = ArduinoEvent;
 55 
 56 	return ArduinoEvent;
 57 
 58 }());
