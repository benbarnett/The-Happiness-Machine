  4 
  5 ARDJS.namespace('ARDJS.ui.BlinkM');
  6 
  7 ARDJS.ui.BlinkM = (function() {
  8 
  9 	var BlinkM;
 10 
 11 	// dependencies
 12 	var I2CBase = ARDJS.I2CBase;
 13 
 14 	/**
 15 	 * Creates and BlinkM object
 16 	 *
 17 	 * @exports BlinkM as ARDJS.ui.BlinkM
 18 	 * @constructor
 19 	 * @augments ARDJS.I2CBase 
 20 	 * @param {Arduino} board The Arduino instance
 21 	 * @param {Number} address The i2c address of the BlinkM module
 22 	 */
 23 	BlinkM = function(board, address) {
 24 
 25 		address = address || 0x09;	// default i2c address for BlinkM
 26 
 27 		this.name = "BlinkM"; // for testing
 28 		
 29 		// call super class
 30 		I2CBase.call(this, board, address);
 31 		
 32 
 33 	};
 34 
 35 	BlinkM.prototype = ARDJS.inherit(I2CBase.prototype);
 36 	BlinkM.prototype.constructor = BlinkM;
 37 
 38 
 39 	/**
 40 	 * Sets the BlinkM to the specified RGB color immediately
 41 	 * @param {Number{}} color An array containing the RGB values. 
 42 	 * color[0] = R, color[1] = G, color[2] = B
 43 	 */
 44 	BlinkM.prototype.goToRGBColorNow = function(color) {
 45 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x6E, color[0], color[1], color[2]]);
 46 	};
 47 	
 48 	/**
 49 	 * Fades to the specified RGB color in the specified time duration. 
 50 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 51 	 *
 52 	 * @param {Number[]} color An array containing the RGB values.
 53 	 * color[0] = R, color[1] = G, color[2] = B
 54 	 * @param {Number} speed The fade speed. Default value is 15.
 55 	 */
 56 	BlinkM.prototype.fadeToRGBColor = function(color, speed) {
 57 		var speed = speed || -1;
 58 		if (speed >= 0) {
 59 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
 60 		}
 61 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x63, color[0], color[1], color[2]]);
 62 	};
 63 
 64 	/**
 65 	 * Fade to a random RGB color.
 66 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 67 	 *
 68 	 * @param {Number[]} colorRange An array containing a range for each color value.
 69 	 * colorRange[0] = range for Red (0-255), colorRange[1] = range for Green, etc.
 70 	 * @param {Number} speed The fade speed. Default value is 15.
 71 	 */
 72 	BlinkM.prototype.fadeToRandomRGBColor = function (colorRange, speed) {
 73 		var speed = speed || -1;
 74 		if (speed >= 0) {
 75 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
 76 		}
 77 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x43, colorRange[0], colorRange[1], colorRange[2]]);
 78 	};
 79 
 80 	/**
 81 	 * Fades to the specified HSB color in the specified time duration. 
 82 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 83 	 *
 84 	 * @param {Number[]} color An array containing the HSB values.
 85 	 * color[0] = H, color[1] = S, color[2] = B
 86 	 * @param {Number} speed The fade speed. Default value is 15.
 87 	 */
 88 	BlinkM.prototype.fadeToHSBColor = function(color, speed) {
 89 		var speed = speed || -1;
 90 		if (speed >= 0) {
 91 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
 92 		}
 93 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x68, color[0], color[1], color[2]]);
 94 	};
 95 	
 96 	/**
 97 	 * Fade to a random HSB color.
 98 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 99 	 *
100 	 * @param {Number[]} colorRange An array containing a range for each color value.
101 	 * colorRange[0] = range for Hue (0-255), colorRange[1] = range for Saturation, etc.
102 	 * @param {Number} speed The fade speed. Default value is 15.
103 	 */	
104 	BlinkM.prototype.fadeToRandomHSBColor = function(colorRange, speed) {
105 		var speed = speed || -1;
106 		if (speed >= 0) {
107 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
108 		}
109 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x48, colorRange[0], colorRange[1], colorRange[2]]);
110 	};
111 
112 	/**
113 	 * Set the rate at which color fading happens. The range is from 1 to 255, where 1 is the
114 	 * slowest and 255 is the fastest (immediate).
115 	 *
116 	 * @param {Number} speed
117 	 */
118 	BlinkM.prototype.setFadeSpeed = function(speed) {
119 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
120 	};
121 	
122 	/**
123 	 * Play a predefined light script. See the BlinkM datasheet page 20 for a list and 
124 	 * description of the predefined scripts.
125 	 *
126 	 * @param {Number} scriptId The id of the light script (from 0 to 18).
127 	 * @param {Number} theNumberOfRepeats The number of times the script should repeat.
128 	 * @param {Number} lineNumber The line number to begin the script from.
129 	 */
130 	BlinkM.prototype.playLightScript = function(scriptId, theNumberOfRepeats, lineNumber) {
131 		var theNumberOfRepeats = theNumberOfRepeats || 1;
132 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x70, scriptId, theNumberOfRepeats, lineNumber]);
133 	};
134 
135 	/**
136 	 * Stop the currently playing predefined light script.
137 	 */
138 	BlinkM.prototype.stopScript = function() {
139 		//self.sendI2CRequest([I2CBase.WRITE, this.address, 'o'.charCodeAt(0)]);
140 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x6F]);
141 	};
142 
143 	/**
144 	 * @private
145 	 */
146 	BlinkM.prototype.handleI2C = function(data) {
147 		// TODO: implement if needed
148 		console.log("BlinkM: " + data);
149 	};
150 
151 	return BlinkM;
152 
153 }());
